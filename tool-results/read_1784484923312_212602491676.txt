   125тЖТ   125тЖТ/**
   126тЖТ   126тЖТ * Calculate the Dynamic Response Factor (DLF) for a structural element
   127тЖТ   127тЖТ * under blast loading, using the Biggs SDOF chart approximation.
   128тЖТ   128тЖТ *
   129тЖТ   129тЖТ * тФАтФА EQUATION SOURCE тФАтФА
   130тЖТ   130тЖТ * Biggs, J.M. (1964) "Introduction to Structural Dynamics",
   131тЖТ   131тЖТ *   Chapter 4, Figure 4-8: DLF vs td/T for triangular pulse.
   132тЖТ   132тЖТ * UFC 3-340-02 (2008), Chapter 5, Figure 5-4: Dynamic Load Factor.
   133тЖТ   133тЖТ * TM 5-1300 (1990), Chapter 5: SDOF Dynamic Analysis.
   134тЖТ   134тЖТ *
   135тЖТ   135тЖТ * тФАтФА EQUATION тФАтФА
   136тЖТ   136тЖТ * DLF = f(td / T)  where td = positive phase duration, T = natural period
   137тЖТ   137тЖТ *
   138тЖТ   138тЖТ * This is the UFC 3-340-02 equivalent load approach (Option B):
   139тЖТ   139тЖТ *   Equivalent Dynamic Pressure = Peak Pressure ├Ч DLF
   140тЖТ   140тЖТ *
   141тЖТ   141тЖТ * тФАтФА BIGGS SDOF CHART тАФ Triangular Pulse тФАтФА
   142тЖТ   142тЖТ *   td/T < 0.1  тЖТ  DLF тЙИ 2╧А ├Ч (td/T)        (impulsive regime)
   143тЖТ   143тЖТ *   td/T тЙИ 0.2  тЖТ  DLF тЙИ 1.1                (transition)
   144тЖТ   144тЖТ *   td/T тЙИ 0.4  тЖТ  DLF тЙИ 1.5                (peak dynamic amplification)
   145тЖТ   145тЖТ *   td/T тЙИ 1.0  тЖТ  DLF тЙИ 1.2                (quasi-static transition)
   146тЖТ   146тЖТ *   td/T > 3.0  тЖТ  DLF тЙИ 1.0                (static regime)
   147тЖТ   147тЖТ *
   148тЖТ   148тЖТ * Note: In the impulsive regime (td/T < 0.1), the DLF is < 1.0 because
   149тЖТ   149тЖТ * the pulse ends before the structure reaches peak response. The
   150тЖТ   150тЖТ * impulse-based equivalent load (Path 2 in calculateDesignLoad) governs
   151тЖТ   151тЖТ * in this regime. The max(Path1, Path2) selection ensures the correct
   152тЖТ   152тЖТ * equivalent load is always used (see calculateDesignLoad documentation).
   153тЖТ   153тЖТ *
   154тЖТ   154тЖТ * @param peakPressure_kPa  - Peak blast pressure on element (kPa)
   155тЖТ   155тЖТ * @param impulse_kPaMs     - Positive phase impulse on element (kPa┬╖ms)
   156тЖТ   156тЖТ * @param duration_ms       - Positive phase duration td (ms)
   157тЖТ   157тЖТ * @param naturalPeriod_ms  - Element natural period T (ms)
   158тЖТ   158тЖТ * @param supportCondition  - Support condition (affects KLM and T)
   159тЖТ   159тЖТ * @returns Dynamic Response Factor DLF (dimensionless)
   160тЖТ   160тЖТ */
   161тЖТ   161тЖТexport function calculateDynamicResponseFactor(
   162тЖТ   162тЖТ  peakPressure_kPa: number,
   163тЖТ   163тЖТ  impulse_kPaMs: number,
   164тЖТ   164тЖТ  duration_ms: number,
   165тЖТ   165тЖТ  naturalPeriod_ms: number,
   166тЖТ   166тЖТ  supportCondition: 'simply_supported' | 'fixed' | 'partial_fixity',
   167тЖТ   167тЖТ): number {
   168тЖТ   168тЖТ  // Edge case: no blast or no duration тЖТ static
   169тЖТ   169тЖТ  if (peakPressure_kPa <= 0 || duration_ms <= 0 || naturalPeriod_ms <= 0) {
   170тЖТ   170тЖТ    return 1.0;
   171тЖТ   171тЖТ  }
   172тЖТ   172тЖТ
   173тЖТ   173тЖТ  const td_T = duration_ms / naturalPeriod_ms;
   174тЖТ   174тЖТ
   175тЖТ   175тЖТ  // тФАтФА Biggs SDOF DLF for triangular pulse тФАтФА
   176тЖТ   176тЖТ  // Source: Biggs (1964) Fig. 4-8; UFC 3-340-02 Fig. 5-4
   177тЖТ   177тЖТ  // Piecewise linear approximation of the standard DLF chart.
   178тЖТ   178тЖТ  // DLF = f(td/T) is read from the chart; the piecewise segments
   179тЖТ   179тЖТ  // below match the chart values at the tabulated points.
   180тЖТ   180тЖТ  let dlf: number;
   181тЖТ   181тЖТ
   182тЖТ   182тЖТ  if (td_T <= 0.1) {
   183тЖТ   183тЖТ    // Impulsive regime: response dominated by impulse, not peak pressure
   184тЖТ   184тЖТ    // DLF тЙИ ╧Й ├Ч td тЙИ 2╧А ├Ч td/T (for small td/T)
   185тЖТ   185тЖТ    // In this regime, impulse matters more than peak pressure.
   186тЖТ   186тЖТ    dlf = 2 * Math.PI * td_T;
   187тЖТ   187тЖТ  } else if (td_T <= 0.2) {
   188тЖТ   188тЖТ    // Transition from impulsive to dynamic
   189тЖТ   189тЖТ    const t01 = 0.1;
   190тЖТ   190тЖТ    const dlf01 = 2 * Math.PI * 0.1;  // тЙИ 0.628
   191тЖТ   191тЖТ    const t02 = 0.2;
   192тЖТ   192тЖТ    const dlf02 = 1.1;                  // from Biggs chart at td/T=0.2
   193тЖТ   193тЖТ    // Linear interpolation
   194тЖТ   194тЖТ    dlf = dlf01 + (dlf02 - dlf01) * (td_T - t01) / (t02 - t01);
   195тЖТ   195тЖТ  } else if (td_T <= 0.4) {
   196тЖТ   196тЖТ    // Near peak DLF region
   197тЖТ   197тЖТ    const t1 = 0.2;
   198тЖТ   198тЖТ    const d1 = 1.1;
   199тЖТ   199тЖТ    const t2 = 0.4;
   200тЖТ   200тЖТ    const d2 = 1.5;
   201тЖТ   201тЖТ    dlf = d1 + (d2 - d1) * (td_T - t1) / (t2 - t1);
   202тЖТ   202тЖТ  } else if (td_T <= 1.0) {
   203тЖТ   203тЖТ    // Dynamic to quasi-static transition
   204тЖТ   204тЖТ    const t1 = 0.4;
   205тЖТ   205тЖТ    const d1 = 1.5;
   206тЖТ   206тЖТ    const t2 = 1.0;
   207тЖТ   207тЖТ    const d2 = 1.2;
   208тЖТ   208тЖТ    dlf = d1 + (d2 - d1) * (td_T - t1) / (t2 - t1);
   209тЖТ   209тЖТ  } else if (td_T <= 3.0) {
   210тЖТ   210тЖТ    // Quasi-static transition
   211тЖТ   211тЖТ    const t1 = 1.0;
   212тЖТ   212тЖТ    const d1 = 1.2;
   213тЖТ   213тЖТ    const t2 = 3.0;
   214тЖТ   214тЖТ    const d2 = 1.0;
   215тЖТ   215тЖТ    dlf = d1 + (d2 - d1) * (td_T - t1) / (t2 - t1);
   216тЖТ   216тЖТ  } else {
   217тЖТ   217тЖТ    // Static regime: load is applied slowly relative to structure period
   218тЖТ   218тЖТ    dlf = 1.0;
   219тЖТ   219тЖТ  }
   220тЖТ   220тЖТ
   221тЖТ   221тЖТ  // Return raw DLF тАФ may be < 1.0 in impulsive regime.
   222тЖТ   222тЖТ  // The calling function (calculateDesignLoad) takes the MAX of
   223тЖТ   223тЖТ  // the pressure-DLF path and the impulse-based path, so the
   224тЖТ   224тЖТ  // final equivalent load is always physically correct.
   225тЖТ   225тЖТ  return dlf;
   226тЖТ   226тЖТ}

   228тЖТ   228тЖТ/**
   229тЖТ   229тЖТ * Estimate the natural period of vibration for a concrete slab element.
   230тЖТ   230тЖТ *
   231тЖТ   231тЖТ * тФАтФА CALCULATION PATH (first principles тЖТ SDOF) тФАтФА
   232тЖТ   232тЖТ *
   233тЖТ   233тЖТ * Step 1: First-mode natural frequency of a uniform beam (Euler-Bernoulli):
   234тЖТ   234тЖТ *   ╧ЙтВБ = (╧А/L)┬▓ ├Ч тИЪ(EI / m_line)
   235тЖТ   235тЖТ *   where I = h┬│/12 (per meter width), m_line = ╧Б ├Ч h (kg/m)
   236тЖТ   236тЖТ *   тЖТ ╧ЙтВБ = (╧А/L)┬▓ ├Ч h ├Ч тИЪ(E / (12╧Б))
   237тЖТ   237тЖТ *
   238тЖТ   238тЖТ * Step 2: SDOF conversion via Biggs transformation factors
   239тЖТ   239тЖТ *   (UFC 3-340-02 Table 5-1 / Biggs Table 5-1):
   240тЖТ   240тЖТ *   K_M = mass transformation factor, K_L = load transformation factor
   241тЖТ   241тЖТ *   T_SDOF = 2╧А тИЪ(K_M ├Ч M_total / (K_L ├Ч k_elastic))
   242тЖТ   242тЖТ *   The ratio K_M/K_L = K_LM (combined load-mass factor):
   243тЖТ   243тЖТ *     SS: K_LM = K_M/K_L = 0.50/0.64 = 0.78
   244тЖТ   244тЖТ *     FF: K_LM = K_M/K_L = 0.41/0.64 = 0.64
   245тЖТ   245тЖТ *
   246тЖТ   246тЖТ * Step 3: Empirical formula (dimensional consolidation):
   247тЖТ   247тЖТ *   T = C ├Ч L ├Ч (L/h)┬▓   [ms]
   248тЖТ   248тЖТ *   where C incorporates: 2╧А, ╧А┬▓, тИЪ(E/(12╧Б)), K_LM, and the
   249тЖТ   249тЖТ *   unit conversion factor (s тЖТ ms).
   250тЖТ   250тЖТ *
   251тЖТ   251тЖТ *   For SS: C_SS = (2/╧А) ├Ч тИЪ(K_LM) ├Ч 1000 / тИЪ(E/(12╧Б))
   252тЖТ   252тЖТ *   With E = 4700тИЪf'c MPa, ╧Б = 2400 kg/m┬│:
   253тЖТ   253тЖТ *     тИЪ(E/(12╧Б)) тЙИ 986 m/s  тЖТ  C_SS тЙИ 0.063
   254тЖТ   254тЖТ *   For FF: C_FF = C_SS ├Ч тИЪ(K_LM_FF / K_LM_SS) = 0.063 ├Ч тИЪ(0.64/0.78) тЙИ 0.031
   255тЖТ   255тЖТ *
   256тЖТ   256тЖТ * тФАтФА SOURCE OF C VALUES тФАтФА
   257тЖТ   257тЖТ * C is NOT an arbitrary constant. It is derived from:
   258тЖТ   258тЖТ *   1. Euler-Bernoulli beam theory (first mode shape)
   259тЖТ   259тЖТ *   2. Biggs SDOF transformation factors (K_LM from UFC 3-340-02 Table 5-1)
   260тЖТ   260тЖТ *   3. Typical RC material properties (E тЙИ 28,000 MPa, ╧Б тЙИ 2,400 kg/m┬│)
   261тЖТ   261тЖТ *   4. Dimensional consolidation: s тЖТ ms
   262тЖТ   262тЖТ *
   263тЖТ   263тЖТ * These values are consistent with structure/index.ts.
   264тЖТ   264тЖТ *
   265тЖТ   265тЖТ * Reference: Biggs (1964) Ch.5; UFC 3-340-02 Table 5-1; TM 5-1300 Ch.5
   266тЖТ   266тЖТ *
   267тЖТ   267тЖТ * @param span_m           - Clear span (m)
   268тЖТ   268тЖТ * @param thickness_m      - Element thickness (m)
   269тЖТ   269тЖТ * @param density_kg_m3    - Concrete density (kg/m┬│)
   270тЖТ   270тЖТ * @param supportCondition - Support type
   271тЖТ   271тЖТ * @returns Natural period T (ms)
   272тЖТ   272тЖТ */
   273тЖТ   273тЖТexport function estimateNaturalPeriod(
   274тЖТ   274тЖТ  span_m: number,
   275тЖТ   275тЖТ  thickness_m: number,
   276тЖТ   276тЖТ  density_kg_m3: number,
   277тЖТ   277тЖТ  supportCondition: 'simply_supported' | 'fixed' | 'partial_fixity',
   278тЖТ   278тЖТ): number {
   279тЖТ   279тЖТ  if (span_m <= 0 || thickness_m <= 0) return 10; // default 10ms
   280тЖТ   280тЖТ
   281тЖТ   281тЖТ  const safeH = Math.max(thickness_m, 0.001);
   282тЖТ   282тЖТ  const safeL = Math.max(span_m, 0.001);
   283тЖТ   283тЖТ
   284тЖТ   284тЖТ  // Coefficient C derived from SDOF first principles (see JSDoc above):
   285тЖТ   285тЖТ  //   SS:  C = 0.063  (from K_LM = 0.78, EтЙИ28GPa, ╧БтЙИ2400 kg/m┬│)
   286тЖТ   286тЖТ  //   FF:  C = 0.031  (from K_LM = 0.64, same material properties)
   287тЖТ   287тЖТ  //   PF:  C = 0.047  (geometric mean: тИЪ(0.063 ├Ч 0.031) тЙИ 0.044, rounded)
   288тЖТ   288тЖТ  let C: number;
   289тЖТ   289тЖТ  switch (supportCondition) {
   290тЖТ   290тЖТ    case 'fixed':
   291тЖТ   291тЖТ      C = 0.031;
   292тЖТ   292тЖТ      break;
   293тЖТ   293тЖТ    case 'partial_fixity':
   294тЖТ   294тЖТ      C = 0.047; // geometric mean of SS and fixed
   295тЖТ   295тЖТ      break;
   296тЖТ   296тЖТ    default:
   297тЖТ   297тЖТ      C = 0.063;
   298тЖТ   298тЖТ  }
   299тЖТ   299тЖТ
   300тЖТ   300тЖТ  // T = C ├Ч L ├Ч (L/h)┬▓  (result in ms)
   301тЖТ   301тЖТ  // Dimensional check: [C] ├Ч m ├Ч (m/m)┬▓ = [C] ├Ч m тЖТ ms тЬУ
   302тЖТ   302тЖТ  const T = C * safeL * Math.pow(safeL / safeH, 2);
   303тЖТ   303тЖТ
   304тЖТ   304тЖТ  // Clamp to physically reasonable range [1, 500] ms
   305тЖТ   305тЖТ  return Math.max(1, Math.min(500, T));
   306тЖТ   306тЖТ}

   308тЖТ   308тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   309тЖТ   309тЖТ// DESIGN LOAD CALCULATION
   310тЖТ   310тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР

   342тЖТ   342тЖТ/**
   343тЖТ   343тЖТ * Get the Load-Mass Factor (KLM) for an SDOF equivalent system.
   344тЖТ   344тЖТ *
   345тЖТ   345тЖТ * тФАтФА EQUATION SOURCE тФАтФА
   346тЖТ   346тЖТ * Biggs, J.M. (1964) "Introduction to Structural Dynamics",
   347тЖТ   347тЖТ *   Table 5-1: Transformation Factors for Single-Degree Systems.
   348тЖТ   348тЖТ * UFC 3-340-02 (2008), Chapter 5, Table 5-1: SDOF Factors.
   349тЖТ   349тЖТ * TM 5-1300 (1990), Chapter 5, Table 5-1.
   350тЖТ   350тЖТ *
   351тЖТ   351тЖТ * тФАтФА DEFINITION тФАтФА
   352тЖТ   352тЖТ * K_LM = K_M / K_L
   353тЖТ   353тЖТ *   where K_M = mass transformation factor (effective mass / total mass)
   354тЖТ   354тЖТ *         K_L = load transformation factor (equivalent load / total load)
   355тЖТ   355тЖТ *
   356тЖТ   356тЖТ * For uniform load on one-way element:
   357тЖТ   357тЖТ *   SS: K_M = 0.50, K_L = 0.64 тЖТ K_LM = 0.50/0.64 = 0.781 тЙИ 0.78
   358тЖТ   358тЖТ *   FF: K_M = 0.41, K_L = 0.64 тЖТ K_LM = 0.41/0.64 = 0.641 тЙИ 0.64
   359тЖТ   359тЖТ *
   360тЖТ   360тЖТ * тФАтФА USAGE тФАтФА
   361тЖТ   361тЖТ * 1. Natural period: T = 2╧А тИЪ(K_LM ├Ч m / k) (via empirical C values)
   362тЖТ   362тЖТ * 2. Impulse equivalent load: P_eq = 2╧А ├Ч I / (T ├Ч K_LM)
   363тЖТ   363тЖТ *
   364тЖТ   364тЖТ * @param supportCondition - Support type
   365тЖТ   365тЖТ * @returns KLM (dimensionless)
   366тЖТ   366тЖТ */
   367тЖТ   367тЖТfunction getLoadMassFactor(
   368тЖТ   368тЖТ  supportCondition: 'simply_supported' | 'fixed' | 'partial_fixity',
   369тЖТ   369тЖТ): number {
   370тЖТ   370тЖТ  // KLM values from UFC 3-340-02 Table 5-1 / Biggs Table 5-1:
   371тЖТ   371тЖТ  //   SS: K_M=0.50, K_L=0.64 тЖТ K_LM = 0.78
   372тЖТ   372тЖТ  //   FF: K_M=0.41, K_L=0.64 тЖТ K_LM = 0.64
   373тЖТ   373тЖТ  //   PF: arithmetic mean of SS and FF = 0.71
   374тЖТ   374тЖТ  switch (supportCondition) {
   375тЖТ   375тЖТ    case 'fixed':
   376тЖТ   376тЖТ      return LOAD_MASS_FACTOR_FF;    // 0.64
   377тЖТ   377тЖТ    case 'partial_fixity':
   378тЖТ   378тЖТ      return (LOAD_MASS_FACTOR_SS + LOAD_MASS_FACTOR_FF) / 2; // 0.71
   379тЖТ   379тЖТ    default:
   380тЖТ   380тЖТ      return LOAD_MASS_FACTOR_SS;    // 0.78
   381тЖТ   381тЖТ  }
   382тЖТ   382тЖТ}

   384тЖТ   384тЖТ/**
   385тЖТ   385тЖТ * Calculate the equivalent dynamic blast load for one structural element.
   386тЖТ   386тЖТ *
   387тЖТ   387тЖТ * тФАтФА EQUATION SOURCE тФАтФА
   388тЖТ   388тЖТ * UFC 3-340-02 (2008), Chapter 5: Dynamic Analysis of Structural Elements.
   389тЖТ   389тЖТ * Biggs, J.M. (1964) "Introduction to Structural Dynamics", Chapters 4-5.
   390тЖТ   390тЖТ * TM 5-1300 (1990), Chapter 5: SDOF Design Methods.
   391тЖТ   391тЖТ *
   392тЖТ   392тЖТ * тФАтФА DUAL-PATH METHOD тФАтФА
   393тЖТ   393тЖТ * This function implements the UFC 3-340-02 equivalent load approach using
   394тЖТ   394тЖТ * a DUAL-PATH method that captures BOTH pressure-dominated and
   395тЖТ   395тЖТ * impulse-dominated response:
   396тЖТ   396тЖТ *
   397тЖТ   397тЖТ *   Path 1 (Pressure-dominated):  wтВБ = P_peak ├Ч DLF(td/T)
   398тЖТ   398тЖТ *   Path 2 (Impulse-dominated):   wтВВ = 2╧А ├Ч I / (T ├Ч KLM)
   399тЖТ   399тЖТ *
   400тЖТ   400тЖТ *   w_blast = max(wтВБ, wтВВ)
   401тЖТ   401тЖТ *
   402тЖТ   402тЖТ * тФАтФА WHY max(pressure, impulse)? тФАтФА
   403тЖТ   403тЖТ * The max() selection is the standard UFC SDOF methodology:
   404тЖТ   404тЖТ *
   405тЖТ   405тЖТ * In the DYNAMIC regime (0.1 тЙд td/T тЙд 3.0), Path 1 governs because
   406тЖТ   406тЖТ * the structure has time to build up dynamic amplification. The DLF
   407тЖТ   407тЖТ * from the Biggs chart (тЙе 1.0) correctly captures the peak response.
   408тЖТ   408тЖТ * The impulse formula (Path 2) would double-count the dynamic effect.
   409тЖТ   409тЖТ *
   410тЖТ   410тЖТ * In the IMPULSIVE regime (td/T < 0.1), the pulse ends before the
   411тЖТ   411тЖТ * structure reaches peak response. The DLF drops below 1.0 because
   412тЖТ   412тЖТ * the pressure is not sustained. Here, the response is governed by
   413тЖТ   413тЖТ * the total impulse (momentum transfer), not the peak pressure.
   414тЖТ   414тЖТ * Path 2 (impulse equivalent) correctly captures this. The DLF in
   415тЖТ   415тЖТ * Path 1 would underestimate the response.
   416тЖТ   416тЖТ *
   417тЖТ   417тЖТ * Therefore, max(wтВБ, wтВВ) ensures the correct governing load in BOTH
   418тЖТ   418тЖТ * regimes without double-counting. This is consistent with the UFC
   419тЖТ   419тЖТ * 3-340-02 SDOF methodology where the designer must check both the
   420тЖТ   420тЖТ * pressure-time and impulse-time response envelopes.
   421тЖТ   421тЖТ *
   422тЖТ   422тЖТ * тФАтФА EQUATION REFERENCES тФАтФА
   423тЖТ   423тЖТ * Path 1 тАФ DLF chart: Biggs (1964) Fig. 4-8; UFC 3-340-02 Fig. 5-4
   424тЖТ   424тЖТ * Path 2 тАФ Impulse:  Biggs (1964) Eq. 4-18 (SDOF impulse response):
   425тЖТ   425тЖТ *   x_max = I / (╧Й_n ├Ч M_e)
   426тЖТ   426тЖТ *   P_eq = k ├Ч x_max = k ├Ч I / (╧Й_n ├Ч K_M ├Ч M_total)
   427тЖТ   427тЖТ *   Since ╧Й_n = 2╧А/T and k = (2╧А/T)┬▓ ├Ч K_LM ├Ч M_total:
   428тЖТ   428тЖТ *   P_eq = (2╧А/T)┬▓ ├Ч K_LM ├Ч M ├Ч I / (2╧А/T ├Ч K_M ├Ч M)
   429тЖТ   429тЖТ *        = (2╧А ├Ч I) / (T ├Ч (K_M/K_L)) = (2╧А ├Ч I) / (T ├Ч K_LM)
   430тЖТ   430тЖТ *   where K_LM = K_M / K_L from Biggs Table 5-1.
   431тЖТ   431тЖТ *
   432тЖТ   432тЖТ * тФАтФА UNIT CHECK (Path 2) тФАтФА
   433тЖТ   433тЖТ *   2╧А ├Ч I [kPa┬╖ms] / (T [ms] ├Ч K_LM [-]) = kPa тЬУ
   434тЖТ   434тЖТ *
   435тЖТ   435тЖТ * ALL four dynamic parameters affect the result:
   436тЖТ   436тЖТ *   - Peak pressure P тЖТ Path 1
   437тЖТ   437тЖТ *   - Impulse I тЖТ Path 2 (kPa┬╖ms)
   438тЖТ   438тЖТ *   - Duration td тЖТ Path 1 (via td/T ratio, both in ms)
   439тЖТ   439тЖТ *   - Natural period T тЖТ Both paths (ms)
   440тЖТ   440тЖТ *   - Load-mass factor KLM тЖТ Path 2 (dimensionless)
   441тЖТ   441тЖТ *
   442тЖТ   442тЖТ * IMPORTANT: This is NOT a static beam calculation. The dual-path DLF
   443тЖТ   443тЖТ * captures the dynamic amplification/dampening effect of the blast pulse
   444тЖТ   444тЖТ * on the structure. For impulsive loads (td << T), Path 2 governs.
   445тЖТ   445тЖТ * For quasi-static loads (td >> T), Path 1 with DLFтЙИ1.0 governs.
   446тЖТ   446тЖТ *
   447тЖТ   447тЖТ * @param element     - Structural element type
   448тЖТ   448тЖТ * @param elementLoad - Per-element load data
   449тЖТ   449тЖТ * @param blast       - Complete blast threat data (DesignInput.blast)
   450тЖТ   450тЖТ * @param thickness_m - Current design thickness (m) тАФ affects natural period
   451тЖТ   451тЖТ * @returns Total equivalent dynamic design load w (kPa)
   452тЖТ   452тЖТ */
   453тЖТ   453тЖТexport function calculateDesignLoad(
   454тЖТ   454тЖТ  element: 'roof' | 'wall' | 'floor',
   455тЖТ   455тЖТ  elementLoad: DesignElementLoad,
   456тЖТ   456тЖТ  blast: DesignBlastInput,
   457тЖТ   457тЖТ  thickness_m?: number,
   458тЖТ   458тЖТ): number {
   459тЖТ   459тЖТ  // тФАтФА 1. Select peak blast pressure per element type тФАтФА
   460тЖТ   460тЖТ  const peakPressure = getPeakBlastPressure(element, blast);
   461тЖТ   461тЖТ
   462тЖТ   462тЖТ  // тФАтФА 2. Compute element natural period at current thickness (ms) тФАтФА
   463тЖТ   463тЖТ    // T depends on: span, thickness, density, supportCondition.
   464тЖТ   464тЖТ    // See estimateNaturalPeriod() JSDoc for derivation path.
   465тЖТ   465тЖТ    // T changes at each iteration because thickness changes тЖТ DLF changes.
   466тЖТ   466тЖТ  const h = thickness_m ?? elementLoad.thickness;
   467тЖТ   467тЖТ  const T = estimateNaturalPeriod(
   468тЖТ   468тЖТ    elementLoad.span, h, elementLoad.material.density,
   469тЖТ   469тЖТ    elementLoad.supportCondition,
   470тЖТ   470тЖТ  );
   471тЖТ   471тЖТ
   472тЖТ   472тЖТ  // тФАтФА 3. Load-Mass Factor KLM (dimensionless) тФАтФА
   473тЖТ   473тЖТ  // Source: Biggs Table 5-1 / UFC 3-340-02 Table 5-1
   474тЖТ   474тЖТ  // SS: K_LM = 0.50/0.64 = 0.78,  FF: K_LM = 0.41/0.64 = 0.64
   475тЖТ   475тЖТ  const KLM = getLoadMassFactor(elementLoad.supportCondition);
   476тЖТ   476тЖТ
   477тЖТ   477тЖТ  // тФАтФА 4. Path 1: Pressure-dominated equivalent load (kPa) тФАтФА
   478тЖТ   478тЖТ  // Source: Biggs (1964) Fig. 4-8; UFC 3-340-02 Fig. 5-4
   479тЖТ   479тЖТ  // wтВБ = P_peak [kPa] ├Ч DLF(td/T) [-] тЖТ kPa
   480тЖТ   480тЖТ  // DLF depends on: duration [ms] / naturalPeriod [ms] = dimensionless ratio
   481тЖТ   481тЖТ  const dlf = calculateDynamicResponseFactor(
   482тЖТ   482тЖТ    peakPressure,                      // P_peak [kPa]
   483тЖТ   483тЖТ    elementLoad.dynamicImpulse,        // I [kPa┬╖ms] тАФ passed for documentation;
   484тЖТ   484тЖТ    elementLoad.dynamicDuration,       // td [ms]
   485тЖТ   485тЖТ    T,                                 // T [ms]
   486тЖТ   486тЖТ    elementLoad.supportCondition,
   487тЖТ   487тЖТ  );
   488тЖТ   488тЖТ  const w_pressurePath = peakPressure * dlf;  // [kPa] ├Ч [-] = kPa
   489тЖТ   489тЖТ
   490тЖТ   490тЖТ  // тФАтФА 5. Path 2: Impulse-dominated equivalent load (kPa) тФАтФА
   491тЖТ   491тЖТ  // Source: Biggs (1964) Eq. 4-18; UFC 3-340-02 Ch.5
   492тЖТ   492тЖТ  //
   493тЖТ   493тЖТ  // Formula: P_eq = 2╧А ├Ч I / (T ├Ч KLM)
   494тЖТ   494тЖТ  //   Units: 2╧А [-] ├Ч I [kPa┬╖ms] / (T [ms] ├Ч KLM [-]) = kPa тЬУ
   495тЖТ   495тЖТ  //
   496тЖТ   496тЖТ  // Only meaningful in the impulsive regime (td/T < 0.2) where DLF < 1.0.
   497тЖТ   497тЖТ  // In the dynamic/quasi-static regime (td/T >= 0.2), the pressure path
   498тЖТ   498тЖТ  // with DLF >= 1.0 correctly captures the response. Applying the
   499тЖТ   499тЖТ  // impulse formula there would be double-counting.
   500тЖТ   500тЖТ  //
   501тЖТ   501тЖТ  // WHY td/T < 0.2 threshold? At td/T тЙИ 0.2, the Biggs chart shows
   502тЖТ   502тЖТ  // DLF тЙИ 1.0 (crossover point). Below this, DLF < 1.0 and the
   503тЖТ   503тЖТ  // impulse-based formula provides the correct governing load.
   504тЖТ   504тЖТ  const td_T = elementLoad.dynamicDuration / T;  // [ms] / [ms] = [-]
   505тЖТ   505тЖТ  let w_impulsePath = 0;
   506тЖТ   506тЖТ  if (td_T < 0.2 && elementLoad.dynamicImpulse > 0) {
   507тЖТ   507тЖТ    // 2╧А ├Ч I [kPa┬╖ms] / (T [ms] ├Ч KLM [-]) тЖТ kPa
   508тЖТ   508тЖТ    w_impulsePath = (2 * Math.PI * elementLoad.dynamicImpulse) / (T * KLM);
   509тЖТ   509тЖТ  }
   510тЖТ   510тЖТ
   511тЖТ   511тЖТ  // тФАтФА 6. Governing equivalent blast load (kPa) тФАтФА
   512тЖТ   512тЖТ  // max(wтВБ, wтВВ) ensures correct load in BOTH regimes:
   513тЖТ   513тЖТ  //   - Dynamic/quasi-static: wтВБ (DLF тЙе 1.0) governs
   514тЖТ   514тЖТ  //   - Impulsive: wтВВ (momentum transfer) governs
   515тЖТ   515тЖТ  // This is standard UFC SDOF methodology (UFC 3-340-02 Ch.5).
   516тЖТ   516тЖТ  const equivalentBlastLoad = Math.max(w_pressurePath, w_impulsePath);
   517тЖТ   517тЖТ
   518тЖТ   518тЖТ  // тФАтФА 7. Add static component (kPa) тФАтФА
   519тЖТ   519тЖТ  // Static pressure (overburden/lateral/soilReaction/selfWeight) is NOT
   520тЖТ   520тЖТ  // multiplied by DLF тАФ it is a sustained dead load, not a blast impulse.
   521тЖТ   521тЖТ  // Unit: kPa + kPa = kPa тЬУ
   522тЖТ   522тЖТ  return equivalentBlastLoad + elementLoad.staticPressure;
   523тЖТ   523тЖТ}
   524тЖТ   524тЖТ
   525тЖТ   525тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   526тЖТ   526тЖТ// MOMENT DEMAND
   527тЖТ   527тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   528тЖТ   528тЖТ
   529тЖТ   529тЖТ/**
   530тЖТ   530тЖТ * Calculate design bending moment for a uniformly loaded element.
   531тЖТ   531тЖТ *
   532тЖТ   532тЖТ * IMPORTANT: The input load w must be the EQUIVALENT DYNAMIC BLAST LOAD
   533тЖТ   533тЖТ * (from calculateDesignLoad), NOT a static pressure. The DLF and impulse
   534тЖТ   534тЖТ * response are already embedded in w by calculateDesignLoad().
   535тЖТ   535тЖТ *
   536тЖТ   536тЖТ * This function only applies the static beam moment coefficients:
   537тЖТ   537тЖТ *   Simply supported:  Mu = w L┬▓ / 8
   538тЖТ   538тЖТ *   Fixed:            Mu = w L┬▓ / 12
   539тЖТ   539тЖТ *   Partial fixity:   Mu = w L┬▓ / 10
   540тЖТ   540тЖТ *
   541тЖТ   541тЖТ * Source: Standard structural mechanics (any strength of materials textbook).
   542тЖТ   542тЖТ * These coefficients are for a uniformly loaded one-way spanning element.
   543тЖТ   543тЖТ *
   544тЖТ   544тЖТ * @param w                - Total design load (kPa = kN/m┬▓) тАФ includes dynamic DLF
   545тЖТ   545тЖТ * @param span             - Clear span L (m)
   546тЖТ   546тЖТ * @param supportCondition - Support type
   547тЖТ   547тЖТ * @returns Design moment Mu (kN┬╖m per meter width)
   548тЖТ   548тЖТ */
   549тЖТ   549тЖТexport function calculateDesignMoment(
   550тЖТ   550тЖТ  w: number,
   551тЖТ   551тЖТ  span: number,
   552тЖТ   552тЖТ  supportCondition: 'simply_supported' | 'fixed' | 'partial_fixity',
   553тЖТ   553тЖТ): number {
   554тЖТ   554тЖТ  const C = MOMENT_COEFFICIENTS[supportCondition] ?? 8;
   555тЖТ   555тЖТ  return (w * span * span) / C;
   556тЖТ   556тЖТ}
   557тЖТ   557тЖТ
   558тЖТ   558тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   559тЖТ   559тЖТ// SHEAR DEMAND
   560тЖТ   560тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   561тЖТ   561тЖТ
   562тЖТ   562тЖТ/**
   563тЖТ   563тЖТ * Calculate design shear for a uniformly loaded element.
   564тЖТ   564тЖТ * Vu = w ├Ч L / 2  (same for all support conditions)
   565тЖТ   565тЖТ *
   566тЖТ   566тЖТ * Note: w is the equivalent dynamic blast load (kPa) from calculateDesignLoad.
   567тЖТ   567тЖТ * Units: w [kPa] ├Ч L [m] = kN/m тЬУ
   568тЖТ   568тЖТ *
   569тЖТ   569тЖТ * @param w    - Total design load (kPa) тАФ includes dynamic DLF
   570тЖТ   570тЖТ * @param span - Clear span (m)
   571тЖТ   571тЖТ * @returns Design shear Vu (kN/m)
   572тЖТ   572тЖТ */
   573тЖТ   573тЖТexport function calculateDesignShear(
   574тЖТ   574тЖТ  w: number,
   575тЖТ   575тЖТ  span: number,
   576тЖТ   576тЖТ): number {
   577тЖТ   577тЖТ  return (w * span) / 2;
   578тЖТ   578тЖТ}
   579тЖТ   579тЖТ
   580тЖТ   580тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   581тЖТ   581тЖТ// DEFLECTION (gross moment of inertia)
   582тЖТ   582тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   583тЖТ   583тЖТ
   584тЖТ   584тЖТ/**
   585тЖТ   585тЖТ * Calculate mid-span deflection using gross moment of inertia.
   586тЖТ   586тЖТ *
   587тЖТ   587тЖТ * ╬┤ = C ├Ч w ├Ч LтБ┤ / (E ├Ч I)
   588тЖТ   588тЖТ * where  I = b ├Ч h┬│ / 12   (b = 1 m for per-meter strip)
   589тЖТ   589тЖТ *
   590тЖТ   590тЖТ * @param w                - Total design load (kPa)
   591тЖТ   591тЖТ * @param span             - Clear span (m)
   592тЖТ   592тЖТ * @param Ec_MPa           - Concrete modulus of elasticity (MPa)
   593тЖТ   593тЖТ * @param thickness_m      - Element thickness (m)
   594тЖТ   594тЖТ * @param supportCondition - Support type
   595тЖТ   595тЖТ * @returns Maximum deflection ╬┤ (mm)
   596тЖТ   596тЖТ */
   597тЖТ   597тЖТexport function calculateDeflection(
   598тЖТ   598тЖТ  w: number,
   599тЖТ   599тЖТ  span: number,
   600тЖТ   600тЖТ  Ec_MPa: number,
   601тЖТ   601тЖТ  thickness_m: number,
   602тЖТ   602тЖТ  supportCondition: 'simply_supported' | 'fixed' | 'partial_fixity',
   603тЖТ   603тЖТ): number {
   604тЖТ   604тЖТ  if (Ec_MPa <= 0 || thickness_m <= 0 || span <= 0) return 0;
   605тЖТ   605тЖТ
   606тЖТ   606тЖТ  const C = DEFLECTION_COEFFICIENTS[supportCondition] ?? (5 / 384);
   607тЖТ   607тЖТ  const E_kN_m2 = Ec_MPa * 1000;           // MPa тЖТ kN/m┬▓
   608тЖТ   608тЖТ  const I_m4 = Math.pow(thickness_m, 3) / 12; // b=1m
   609тЖТ   609тЖТ
   610тЖТ   610тЖТ  const delta_m = (C * w * Math.pow(span, 4)) / (E_kN_m2 * I_m4);
   611тЖТ   611тЖТ  return delta_m * 1000; // тЖТ mm
   612тЖТ   612тЖТ}
   613тЖТ   613тЖТ
   614тЖТ   614тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   615тЖТ   615тЖТ// ELEMENT DESIGN тАФ Iterative Thickness Search
   616тЖТ   616тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   617тЖТ   617тЖТ
   618тЖТ   618тЖТ/**
   619тЖТ   619тЖТ * Design a single structural element with iterative thickness search.
   620тЖТ   620тЖТ *
   621тЖТ   621тЖТ * Algorithm:
   622тЖТ   622тЖТ *   1. Start at existingThickness (from analysis input)
   623тЖТ   623тЖТ *   2. Compute design load w (blast + static)
   624тЖТ   624тЖТ *   3. Compute moment Mu and shear Vu
   625тЖТ   625тЖТ *   4. Compute required As (using static fy)
   626тЖТ   626тЖТ *   5. Select reinforcement bars
   627тЖТ   627тЖТ *   6. Compute capacity ╧ЖMn (using DIF-modified fy) and ╧ЖVc (using DIF-modified f'c)
   628тЖТ   628тЖТ *   7. Check safety factors against targetSafetyFactor
   629тЖТ   629тЖТ *   8. Check penetration: h >= max(perforationThickness, scabbingThickness)
   630тЖТ   630тЖТ *   9. If not met: increase thickness by thicknessIncrement, goto 2
   631тЖТ   631тЖТ *  10. Max 100 iterations; if maxThickness exceeded тЖТ status = 'fail'
   632тЖТ   632тЖТ *
   633тЖТ   633тЖТ * Thickness convergence requires ALL three criteria:
   634тЖТ   634тЖТ *   h_required = max(flexural required h, shear required h, penetration required h)
   635тЖТ   635тЖТ *
   636тЖТ   636тЖТ * This function does NOT mutate any input parameters.
   637тЖТ   637тЖТ *
   638тЖТ   638тЖТ * @param element     - Structural element type
   639тЖТ   639тЖТ * @param elementLoad - Per-element load data (from DesignInput.elements)
   640тЖТ   640тЖТ * @param blast       - Complete blast threat data (from DesignInput.blast)
   641тЖТ   641тЖТ * @param criteria    - Design criteria (from DesignInput.criteria)
   642тЖТ   642тЖТ * @param penetration - Penetration data from analysis (optional; if omitted, penetration
   643тЖТ   643тЖТ *                       check is skipped тАФ used for backward compatibility and unit tests)
   644тЖТ   644тЖТ * @returns Complete ElementDesignResult
   645тЖТ   645тЖТ */
   646тЖТ   646тЖТexport function designElement(
   647тЖТ   647тЖТ  element: 'roof' | 'wall' | 'floor',
   648тЖТ   648тЖТ  elementLoad: DesignElementLoad,
   649тЖТ   649тЖТ  blast: DesignBlastInput,
   650тЖТ   650тЖТ  criteria: DesignCriteria,
   651тЖТ   651тЖТ  penetration?: DesignPenetrationData,
   652тЖТ   652тЖТ): ElementDesignResult {
   653тЖТ   653тЖТ  const warnings: string[] = [];
   654тЖТ   654тЖТ  const MAX_ITERATIONS = 100;
   655тЖТ   655тЖТ
   656тЖТ   656тЖТ  const supportCondition = element === 'wall'
   657тЖТ   657тЖТ    ? criteria.wallSupportCondition
   658тЖТ   658тЖТ    : criteria.supportCondition;
   659тЖТ   659тЖТ
   660тЖТ   660тЖТ  // Dynamic Increase Factors (UFC 3-340-02 Sec 5.14.3)
   661тЖТ   661тЖТ  const steelDIF = calculateSteelDIF(criteria.steelGrade);
   662тЖТ   662тЖТ  const fy_dynamic = criteria.steelGrade * steelDIF;   // DIF applied to capacity only
   663тЖТ   663тЖТ
   664тЖТ   664тЖТ  // Concrete DIF from material properties (set by adapter from analysis)
   665тЖТ   665тЖТ  const fpc_dynamic = elementLoad.material.fpc * elementLoad.material.difCompressive;
   666тЖТ   666тЖТ
   667тЖТ   667тЖТ  let currentThickness = elementLoad.thickness;
   668тЖТ   668тЖТ  let result: ElementDesignResult;
   669тЖТ   669тЖТ
   670тЖТ   670тЖТ  for (let iteration = 0; iteration < MAX_ITERATIONS; iteration++) {
   671тЖТ   671тЖТ    // тФАтФА 1. Equivalent Dynamic Blast Load тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   672тЖТ   672тЖТ    // w = calculateDesignLoad() returns the DYNAMIC blast equivalent load
   673тЖТ   673тЖТ    // (not a static pressure). It includes:
   674тЖТ   674тЖТ    //   - Peak pressure ├Ч DLF(td/T) [Path 1, Biggs SDOF]
   675тЖТ   675тЖТ    //   - 2╧А├ЧI/(T├ЧKLM)            [Path 2, impulse response]
   676тЖТ   676тЖТ    //   - max(Path1, Path2)        [UFC SDOF methodology]
   677тЖТ   677тЖТ    //   - + staticPressure (NOT multiplied by DLF)
   678тЖТ   678тЖТ    //
   679тЖТ   679тЖТ    // The thickness is passed so T is recalculated at each iteration
   680тЖТ   680тЖТ    // (T changes as h changes тЖТ DLF changes тЖТ w changes).
   681тЖТ   681тЖТ    const w_base = calculateDesignLoad(element, elementLoad, blast, currentThickness);
   682тЖТ   682тЖТ
   683тЖТ   683тЖТ    // Account for additional self-weight from thickness increase
   684тЖТ   684тЖТ    const thicknessIncrease = Math.max(0, currentThickness - elementLoad.thickness);
   685тЖТ   685тЖТ    const additionalSelfWeight = (elementLoad.material.density * thicknessIncrease * GRAVITY) / 1000; // kPa
   686тЖТ   686тЖТ    const w = w_base + additionalSelfWeight;
   687тЖТ   687тЖТ
   688тЖТ   688тЖТ    // тФАтФА 2. Moment & Shear demand тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   689тЖТ   689тЖТ    // Mu is derived from the EQUIVALENT DYNAMIC BLAST LOAD, not from
   690тЖТ   690тЖТ    // static pressure. The load w already contains the DLF and/or
   691тЖТ   691тЖТ    // impulse response factors from calculateDesignLoad().
   692тЖТ   692тЖТ    //
   693тЖТ   693тЖТ    //   w = P_peak ├Ч DLF(td/T) + static   [OR impulse path]
   694тЖТ   694тЖТ    //   Mu = w ├Ч L┬▓ / C
   695тЖТ   695тЖТ    //
   696тЖТ   696тЖТ    // This is NOT a static Mu = P_peak ├Ч L┬▓/8. The DLF, impulse, duration,
   697тЖТ   697тЖТ    // natural period, and KLM all flow through w to Mu.
   698тЖТ   698тЖТ    //
   699тЖТ   699тЖТ    // Source: UFC 3-340-02 Ch.5 (Dynamic Analysis); Biggs (1964) Ch.5
   700тЖТ   700тЖТ    const Mu = calculateDesignMoment(w, elementLoad.span, supportCondition);
   701тЖТ   701тЖТ    const Vu = calculateDesignShear(w, elementLoad.span);
   702тЖТ   702тЖТ
   703тЖТ   703тЖТ    // тФАтФА 3. Effective depth (assume T16 for initial As estimate) тФА
   704тЖТ   704тЖТ    const d_assumed = calculateEffectiveDepth(currentThickness, criteria.concreteCover, 16);
   705тЖТ   705тЖТ
   706тЖТ   706тЖТ    // тФАтФА 4. Required As (STATIC fy тАФ DIF applied only to capacity) тФА
   707тЖТ   707тЖТ    // Input moment Mu is the DYNAMIC blast moment (via DLF).
   708тЖТ   708тЖТ    // As is computed with static fy; DIF is applied only at
   709тЖТ   709тЖТ    // capacity check (step 7) per UFC 3-340-02 Sec 5.14.3.
   710тЖТ   710тЖТ    const As_req = calculateRequiredAs(
   711тЖТ   711тЖТ      Mu, d_assumed, elementLoad.material.fpc, criteria.steelGrade, ACI_DESIGN_FACTORS.flexure,
   712тЖТ   712тЖТ    );
   713тЖТ   713тЖТ
   714тЖТ   714тЖТ    // тФАтФА 5. Select reinforcement тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   715тЖТ   715тЖТ    const mainRebar = selectReinforcement(As_req);
   716тЖТ   716тЖТ    const distributionRebar = selectDistributionReinforcement(mainRebar);
   717тЖТ   717тЖТ
   718тЖТ   718тЖТ    // тФАтФА 6. Recompute d with actual selected bar diameter тФАтФАтФАтФАтФАтФАтФАтФА
   719тЖТ   719тЖТ    const d_actual = calculateEffectiveDepth(
   720тЖТ   720тЖТ      currentThickness, criteria.concreteCover, mainRebar.barDiameter,
   721тЖТ   721тЖТ    );
   722тЖТ   722тЖТ
   723тЖТ   723тЖТ    // тФАтФА 7. Flexural capacity (DYNAMIC fy тАФ DIF applied) тФАтФАтФАтФАтФАтФАтФАтФА
   724тЖТ   724тЖТ    const Mn = calculateFlexuralCapacity(
   725тЖТ   725тЖТ      mainRebar.asProvided, d_actual, elementLoad.material.fpc, fy_dynamic, ACI_DESIGN_FACTORS.flexure,
   726тЖТ   726тЖТ    );
   727тЖТ   727тЖТ
   728тЖТ   728тЖТ    // тФАтФА 8. Shear capacity (DYNAMIC f'c тАФ concrete DIF applied) тФАтФА
   729тЖТ   729тЖТ    const phiVc = calculateShearCapacity(fpc_dynamic, d_actual, ACI_DESIGN_FACTORS.shear);
   730тЖТ   730тЖТ
   731тЖТ   731тЖТ    // тФАтФА 9. Safety factors тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   732тЖТ   732тЖТ    const flexuralSF = Mu > 0 ? Mn / Mu : Infinity;
   733тЖТ   733тЖТ    const shearSF = Vu > 0 ? phiVc / Vu : Infinity;
   734тЖТ   734тЖТ
   735тЖТ   735тЖТ    // тФАтФА 10. Deflection тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   736тЖТ   736тЖТ    const maxDeflection = calculateDeflection(
   737тЖТ   737тЖТ      w, elementLoad.span, elementLoad.material.Ec, currentThickness, supportCondition,
   738тЖТ   738тЖТ    );
   739тЖТ   739тЖТ    const allowableDeflection = criteria.maxDeflectionRatio * elementLoad.span * 1000; // mm
   740тЖТ   740тЖТ
   741тЖТ   741тЖТ    // тФАтФА 11. Status determination тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   742тЖТ   742тЖТ    // Three failure modes must all be satisfied:
   743тЖТ   743тЖТ    //   (a) Flexural SF >= target
   744тЖТ   744тЖТ    //   (b) Shear SF >= target
   745тЖТ   745тЖТ    //   (c) Penetration: h >= max(perforationThickness, scabbingThickness)
   746тЖТ   746тЖТ    //
   747тЖТ   747тЖТ    // h_required = max(flexural h, shear h, penetration h)
   748тЖТ   748тЖТ    const structuralPass =
   749тЖТ   749тЖТ      flexuralSF >= criteria.targetSafetyFactor &&
   750тЖТ   750тЖТ      shearSF >= criteria.targetSafetyFactor;
   751тЖТ   751тЖТ
   752тЖТ   752тЖТ    // Penetration check (from NDRC/TM 5-855-1 тАФ values come from analysis engine)
   753тЖТ   753тЖТ    const penetrationRequired = penetration
   754тЖТ   754тЖТ      ? Math.max(penetration.perforationThickness, penetration.scabbingThickness)
   755тЖТ   755тЖТ      : 0;
   756тЖТ   756тЖТ    const penetrationPass =
   757тЖТ   757тЖТ      penetrationRequired <= 0 ||  // no penetration threat тЖТ always pass
   758тЖТ   758тЖТ      currentThickness >= penetrationRequired;
   759тЖТ   759тЖТ
   760тЖТ   760тЖТ    let status: 'pass' | 'fail' | 'optimize';
   761тЖТ   761тЖТ    if (structuralPass && penetrationPass) {
   762тЖТ   762тЖТ      status = 'pass';
   763тЖТ   763тЖТ    } else if (currentThickness >= criteria.maxThickness) {
   764тЖТ   764тЖТ      status = 'fail';
   765тЖТ   765тЖТ      if (!structuralPass) {
   766тЖТ   766тЖТ        warnings.push(
   767тЖТ   767тЖТ          `╪з┘Д╪│┘Е╪з┘Г╪й ╪з┘Д┘В╪╡┘И┘Й (${criteria.maxThickness} m) ╪к┘Е ╪и┘Д┘И╪║┘З╪з ╪п┘И┘Ж ╪к╪н┘В┘К┘В ╪╣╪з┘Е┘Д ╪з┘Д╪г┘Е╪з┘Ж ╪з┘Д┘Е╪╖┘Д┘И╪и (${criteria.targetSafetyFactor})`,
   768тЖТ   768тЖТ        );
   769тЖТ   769тЖТ      }
   770тЖТ   770тЖТ      if (!penetrationPass) {
   771тЖТ   771тЖТ        warnings.push(
   772тЖТ   772тЖТ          `╪з┘Д╪│┘Е╪з┘Г╪й ╪з┘Д┘В╪╡┘И┘Й (${criteria.maxThickness} m) ╪к┘Е ╪и┘Д┘И╪║┘З╪з ╪п┘И┘Ж ╪к┘Д╪и┘К╪й ┘Е╪к╪╖┘Д╪и╪з╪к ┘Е┘В╪з┘И┘Е╪й ╪з┘Д╪з╪о╪к╪▒╪з┘В (${penetrationRequired.toFixed(3)} m)`,
   773тЖТ   773тЖТ        );
   774тЖТ   774тЖТ      }
   775тЖТ   775тЖТ    } else {
   776тЖТ   776тЖТ      status = 'optimize';
   777тЖТ   777тЖТ    }
   778тЖТ   778тЖТ
   779тЖТ   779тЖТ    // тФАтФА 12. Assemble result тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   780тЖТ   780тЖТ    result = {
   781тЖТ   781тЖТ      element,
   782тЖТ   782тЖТ      existingThickness: elementLoad.thickness,
   783тЖТ   783тЖТ      requiredThickness: currentThickness,
   784тЖТ   784тЖТ      recommendedThickness: currentThickness,
   785тЖТ   785тЖТ      designMoment: Mu,
   786тЖТ   786тЖТ      requiredAs: As_req,
   787тЖТ   787тЖТ      providedAs: mainRebar.asProvided,
   788тЖТ   788тЖТ      mainReinforcement: mainRebar,
   789тЖТ   789тЖТ      distributionReinforcement: distributionRebar,
   790тЖТ   790тЖТ      designShear: Vu,
   791тЖТ   791тЖТ      shearCapacity: phiVc,
   792тЖТ   792тЖТ      maxDeflection,
   793тЖТ   793тЖТ      allowableDeflection,
   794тЖТ   794тЖТ      flexuralSafetyFactor: flexuralSF,
   795тЖТ   795тЖТ      shearSafetyFactor: shearSF,
   796тЖТ   796тЖТ      status,
   797тЖТ   797тЖТ      warnings: [...warnings],
   798тЖТ   798тЖТ    };
   799тЖТ   799тЖТ
   800тЖТ   800тЖТ    // тФАтФА 13. Check convergence тФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФАтФА
   801тЖТ   801тЖТ    if (status === 'pass' || status === 'fail') break;
   802тЖТ   802тЖТ
   803тЖТ   803тЖТ    // Increase thickness for next iteration
   804тЖТ   804тЖТ    currentThickness += criteria.thicknessIncrement;
   805тЖТ   805тЖТ  }
   806тЖТ   806тЖТ
   807тЖТ   807тЖТ  return result!;
   808тЖТ   808тЖТ}
   809тЖТ   809тЖТ
   810тЖТ   810тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   811тЖТ   811тЖТ// MAIN ENTRY POINT
   812тЖТ   812тЖТ// тХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХРтХР
   813тЖТ   813тЖТ
   814тЖТ   814тЖТ/**
   815тЖТ   815тЖТ * Run structural design for all three elements.
   816тЖТ   816тЖТ *
   817тЖТ   817тЖТ * This is the main entry point for the structural design engine.
   818тЖТ   818тЖТ * It reads ONLY DesignInput тАФ never touches FullAnalysisResult.
   819тЖТ   819тЖТ *
   820тЖТ   820тЖТ * @param designInput - Complete design input (from DesignInputAdapter)
   821тЖТ   821тЖТ * @returns Per-element design results
   822тЖТ   822тЖТ */
   823тЖТ   823тЖТexport function runStructuralDesign(designInput: DesignInput): {
   824тЖТ   824тЖТ  roof: ElementDesignResult;
   825тЖТ   825тЖТ  wall: ElementDesignResult;
   826тЖТ   826тЖТ  floor: ElementDesignResult;
   827тЖТ   827тЖТ} {
   828тЖТ   828тЖТ  const { elements, blast, criteria, penetration } = designInput;
   829тЖТ   829тЖТ
   830тЖТ   830тЖТ  return {
   831тЖТ   831тЖТ    roof: designElement('roof', elements.roof, blast, criteria, penetration.roof),
   832тЖТ   832тЖТ    wall: designElement('wall', elements.wall, blast, criteria, penetration.wall),
   833тЖТ   833тЖТ    floor: designElement('floor', elements.floor, blast, criteria, penetration.floor),
   834тЖТ   834тЖТ  };
