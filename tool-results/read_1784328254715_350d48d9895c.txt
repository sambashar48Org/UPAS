   116→describe('Gate Test 1: Impulse sensitivity', () => {
   117→  it('same Pr=1000 kPa, different impulse → different w_blast', () => {
   118→    const blastA = makeBlast({
   119→      peakReflectedPressure: 1000,
   120→      peakDynamicPressure: 200,

   131→    // Flexible element: T ≈ 0.063 × 6 × (6/0.4)² = 85 ms

   570→

   605→    // Demand uses only:
   606→    //   w = Pr × DLF(T, td, impulse) + staticPressure
   607→    //   Mu = w × L² / C
